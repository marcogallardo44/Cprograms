//

#include "stdafx.h"
#include <cmath>
#include <ctime>
#include <random>
#include <cstdlib>
#include <iostream>
#include <iomanip>
using namespace std;

void generateTime(int&, int&, int&);
void printUniversal(int, int, int);
void printStandard(int, int, int);
bool morning(int, int, int);
bool afterNoon(int, int, int);

int main()
{
	int hour = 0; // hour variable
	int minute = 0; // minute variable
	int second = 0; // second variable

	srand(time(0));

	generateTime(hour, minute, second);

	printUniversal(hour, minute, second);
	printStandard(hour, minute, second);
	
	system("pause");
	return main();
} // end main

void generateTime(int &h, int &m, int &s)
{
	h = rand() % 24 + 0;
	m = rand() % 60 + 0;
	s = rand() % 60 + 0;
}

void printUniversal(int h, int m, int s)
{
	cout << (h < 10 ? "0" : "") << h << ":";
	cout << (m < 10 ? "0" : "") << m << ":";
	cout << (s < 10 ? "0" : "") << s << endl;
}

void printStandard(int h, int m, int s)
{
	cout << (h % 12 < 10 ? "0" : "") << h % 12 << ":";
	cout << (m < 10 ? "0" : "") << m << ":";
	cout << (s < 10 ? "0" : "") << s << " ";
	if (morning(h, m, s))
		cout << "AM" << endl;
	if (afterNoon(h, m, s))
		cout << "PM" << endl;
}

bool morning(int h, int m, int s)
{
	return h < 12;
}

bool afterNoon(int h, int m, int s)
{
	return h >= 12;

}
