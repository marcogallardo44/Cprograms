#include "stdafx.h"
#include <iostream>

using namespace std;

int main()
{
	int x;
	int y;
	int hcd = 0;
	cout << "Please enter 2 numbers: ";
	cin >> x >> y;
	if (y > x)
	{
		int temp = y;
		y = x;
		x = temp;
	}


	for (int i = 1; i <= y; ++i)
	{
		if (x % i == 0 && y % i == 0)
		{
			hcd = i;
		}
	}

	cout << "Highest Common Denominator: " << hcd << endl;


	system("pause");
	return 0;
}
