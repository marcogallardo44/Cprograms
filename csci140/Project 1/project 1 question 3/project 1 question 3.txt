#include "stdafx.h"
#include <iostream>

using namespace std;

int main()
{
	int number;
	int x;
	int remainder;
	int decimal = 0;
	int base = 1;
	

	cout << "Please input a binary number: ";
	cin >> number;
	x = number;
	while (number > 0)
	{
		remainder = number % 10;
		decimal = decimal + remainder * base;
		base = base * 2;
		number = number / 10;
	}

	cout << "The decimal of your input: " << x << " is: " << decimal << endl;


	system("pause");
    return main();
}
