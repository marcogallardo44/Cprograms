//

#include "stdafx.h"
#include <iostream>
using namespace std;

int main()
{
	int x;
	int y;

	cout << "Enter two integers: ";
	cin >> x >> y;

	if (y % x == 0)
	{
		cout << x << " is a multiple of " << y << endl;
	}
	if (y % x != 0)
	{
		cout << x << " is not a multiple of " << y << endl;
	}
	if (x % 2 == 0)
	{
		cout << x << " is an even number." << endl;
	}

	if (x % 2 != 0)
	{
		cout << x << " is an odd number." << endl;
	}



	system("pause");
    return 0;
}
