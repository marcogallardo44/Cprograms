#include "stdafx.h"
#include <iostream>
#include <cmath>

using namespace std;

int input(int age[], int capacity)
{
	const int CAPACITY = 1000;
	age[CAPACITY];
	int current_size = 0;

	cout << "Please enter values, Q to quit:" << endl;
	int input;
	while (cin >> input)
	{
		if (current_size < CAPACITY)
		{
			age[current_size] = input;
			current_size++;
		}
	}



	return current_size++;
}


int people(int age[], int capacity)
{
	int infant = 0;
	int young = 0;
	int middleage = 0;
	int old = 0;
	int older = 0;

	for (int i = 0; i < capacity; i++)
	{
		if (age[i] >= 0 && age[i] <= 18)
			infant++;
		if (age[i] > 18 && age[i] <= 29)
			young++;
		if (age[i] > 29 && age[i] <= 50)
			middleage++;
		if (age[i] > 50 && age[i] <= 69)
			old++;
		if (age[i] > 69)
			older++;
	}

	cout << "Infant workers (0 - 18): " << infant << endl << endl;
	cout << "Young workers (19 - 29): " << young << endl << endl;
	cout << "Middle Aged workers (30 - 50): " << middleage << endl << endl;
	cout << "Old workers (51 - 69): " << old << endl << endl;
	cout << "Older workers (70+): " << older <<  endl << endl;

	

	int average = 0;
	int count = 0;
	for (int i = 0; i < capacity; i++)
	{
		average = average + age[i];

		count++;

	}

	average = average / count;

	cout << "The average of these numbers is: " << average << endl << endl;

	int mode = 0;

	for (int i = 0; i < capacity; i++)
	{
		mode = mode + pow(age[i] - average, 2);
	}
	
	mode = sqrt(mode / count);

	cout << "The standard deviation for these numbers is: " << mode << endl;

	return 0;
}




int main()
{
	const int capacity = 1000;
	int ages[capacity];
	int size = input(ages, capacity);
	int workers = people(ages, size);
	

	cout << workers;
	






	system("pause");
    return 0;
}
