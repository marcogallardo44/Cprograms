#include "stdafx.h"
#include <iostream>
#include <algorithm>

using namespace std;

double sum(double values[], int size)
{
	double total = 0;
	for (int i = 0; i < size; i++)
	{
		total = total + values[i];
	}
	return total;
}

int main()
{
	double values[]= { 32, 54, 67.5, 29, 35, 80, 115, 44.5, 100, 65 };
	int current_size = 10;
	double total = 0;

	total = sum(values, current_size);
	cout << "total is " << total << endl;
	


	system("pause");
	return 0;
}
