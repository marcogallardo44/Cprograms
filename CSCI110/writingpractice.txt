/*
This program computes the volume (in liters) of a six-pack of soda cans and the total volume of a six-pack and two-liter bottle.

Author: Marco Gallardo

Input Variables: none

Process flow: we multiply the cans per pack to the can volume, then we add the bottle volume to get the entire total volume.
total_volume = cans_per_pack*CAN_VOLUME 
total_volume = total_volume + BOTTLE_VOLUME

Output variables: 
total volume = 2.13
total volume = 4.13
*/

#include "stdafx.h"
#include <iostream>

using namespace std;

int main()
{
	int cans_per_pack = 6;
	const double CAN_VOLUME = .355; // Litters in a 12-ounce can
	double total_volume = cans_per_pack*CAN_VOLUME;

	cout << " A six-pack of 12-ounces cans contains "
		<< total_volume << " liters. " << endl;

	const double BOTTLE_VOLUME = 2; // Two-liter bottle

	total_volume = total_volume + BOTTLE_VOLUME;
	cout<< " A sic-pack and a two-litter bottle contain "
		<< total_volume << "liters. " << endl;
	system("pause");
	return 0;
}
